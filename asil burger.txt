# Savat ko'rinishi (inline tugmalar bilan)
def get_cart_inline(user_id):
    cart = user_carts.get(user_id, [])
    if not cart:
        return None, "🛒 *Savat bo'sh*\n\n0 so'm"
    
    builder = InlineKeyboardBuilder()
    
    total = 0
    items_count = {}
    for key in cart:
        items_count[key] = items_count.get(key, 0) + 1
        total += products[key]["price"]
    
    text = "🛒 *SAVAT*\n━━━━━━━━━━━━━━━\n\n"
    
    for key, count in items_count.items():
        product = products[key]
        text += f"{product['emoji']} *{product['name']}*\n"
        text += f"   ✕{count}  =  {product['price'] * count:,} so'm\n"
        text += f"   └ [➖](del_{key})  [➕](add_{key})\n\n"
        
        # Mahsulot boshqaruvi tugmalari
        builder.row(
            InlineKeyboardButton(text=f"➖ {product['name']}", callback_data=f"remove_{key}"),
            InlineKeyboardButton(text=f"➕ {product['name']}", callback_data=f"add_{key}")
        )
    
    text += f"━━━━━━━━━━━━━━━\n💰 *JAMI: {total:,} so'm*"
    
    builder.row(
        InlineKeyboardButton(text="🗑 Savatni tozalash", callback_data="clear_cart"),
        InlineKeyboardButton(text="✅ Buyurtma", callback_data="checkout")
    )
    builder.row(
        InlineKeyboardButton(text="🔙 Menyu", callback_data="main_menu")
    )
    
    return builder.as_markup(), text

@dp.callback_query(lambda c: c.data.startswith("add_"))
async def add_to_cart(callback: types.CallbackQuery):
    user_id = callback.from_user.id
    product_key = callback.data.split("_")[1]
    
    if user_id not in user_phones:
        await callback.answer("Iltimos, avval /start bosing!", show_alert=True)
        return
    
    if user_id not in user_carts:
        user_carts[user_id] = []
    
    user_carts[user_id].append(product_key)
    product = products[product_key]
    cart_count = len(user_carts[user_id])
    
    # Animatsiyali efekt
    await callback.answer(f"✅ {product['name']} qo'shildi! ({cart_count} ta)", show_alert=False)
    
    # Yangilangan menyuni ko'rsatish
    await callback.message.edit_text(
        f"✨ *{product['name']}* savatga qo'shildi!\n\n"
        f"🛒 Savatda: *{cart_count}* ta mahsulot\n"
        f"💰 Jami: *{sum(products[p]['price'] for p in user_carts[user_id]):,}* so'm\n\n"
        f"Yana buyurtma qiling:",
        parse_mode="Markdown",
        reply_markup=get_modern_menu()
    )

@dp.callback_query(lambda c: c.data.startswith("remove_"))
async def remove_from_cart(callback: types.CallbackQuery):
    user_id = callback.from_user.id
    product_key = callback.data.split("_")[1]
    
    if user_id not in user_carts:
        await callback.answer("Savat bo'sh!", show_alert=True)
        return
    
    if product_key in user_carts[user_id]:
        user_carts[user_id].remove(product_key)
        await callback.answer(f"❌ {products[product_key]['name']} o'chirildi")
        
        if not user_carts[user_id]:
            await callback.message.edit_text(
                "🛒 Savat bo'sh",
                reply_markup=get_modern_menu()
            )
        else:
            _, text = get_cart_inline(user_id)
            await callback.message.edit_text(
                text,
                parse_mode="Markdown",
                reply_markup=get_cart_inline(user_id)[0]
            )
    else:
        await callback.answer("Mahsulot topilmadi!", show_alert=True)
